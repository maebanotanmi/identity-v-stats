<script>
(function(){
  const table = document.getElementById('ranking');
  const headers = table.querySelectorAll('thead th');

  function numOrNull(text){
    const t = (text||"").toString().replace(/[^\d.\-]/g,'');
    const v = parseFloat(t);
    return Number.isFinite(v) ? v : null;
  }

  headers.forEach((th, idx) => {
    th.addEventListener('click', () => {
      const type = th.dataset.type || 'string';
      const tbody = table.tBodies[0];
      const rows = Array.from(tbody.rows);
      const asc = !th.classList.contains('asc');

      headers.forEach(h => h.classList.remove('asc','desc'));
      th.classList.toggle('asc', asc);
      th.classList.toggle('desc', !asc);

      rows.sort((a,b) => {
        const A = a.cells[idx].textContent.trim();
        const B = b.cells[idx].textContent.trim();

        if(type === 'number'){
          const aN = numOrNull(A), bN = numOrNull(B);
          if(aN===null && bN===null) return 0;
          if(aN===null) return 1;          // 空欄は後ろへ
          if(bN===null) return -1;
          return asc ? aN - bN : bN - aN;
        }else{
          return asc ? A.localeCompare(B, 'ja') : B.localeCompare(A, 'ja');
        }
      });

      rows.forEach(r => tbody.appendChild(r));
    });
  });
})();
</script>
